referenceFrame = coarseReferenceFrame;
for each in numIterations:
    usefulEyePositionTraces = StripAnalysis(scaledDownInputVideoFrames, referenceFrame);

    cumulativePixelValues = [][];
    counter = [][];

    for each frame in numberOfFrames:
        # Add frame pixel values on to cumulativePixelValues according to its location
        # given by usefulEyePositionTraces.
        cumulativePixelValues[usefulEyePositionTraces[frame]] += frame;

        # Increment counter for pixels affected.
        counter[usefulEyePositionTraces[frame]] += 1;

    referenceFrame = cumulativePixelValues / counter;
